186  void change_capacity(size_type new_capacity) {
187  if (new_capacity <= N) {
188  if (!is_direct()) {
189  T* indirect = indirect_ptr(0);
190  T* src = indirect;
191  T* dst = direct_ptr(0);
192  memcpy(dst, src, size() * sizeof(T));
193  free(indirect);
194  _size -= N + 1;
195  }
196  } else {
197  if (!is_direct()) {
198  /* FIXME: Because malloc/realloc here won't call new_handler if allocation fails, assert
199  success. These should instead use an allocator or new/delete so that handlers
200  are called as necessary, but performance would be slightly degraded by doing so. */
201  _union.indirect = static_cast<char*>(realloc(_union.indirect, ((size_t)sizeof(T)) * new_capacity));
202  assert(_union.indirect);
203  _union.capacity = new_capacity;
204  } else {
205  char* new_indirect = static_cast<char*>(malloc(((size_t)sizeof(T)) * new_capacity));
206  assert(new_indirect);
207  T* src = direct_ptr(0);
208  T* dst = reinterpret_cast<T*>(new_indirect);
209  memcpy(dst, src, size() * sizeof(T));
210  _union.indirect = new_indirect;
211  _union.capacity = new_capacity;
212  _size += N + 1;
213  }
214  }

412  iterator erase(iterator first, iterator last) {
413  // Erase is not allowed to the change the object's capacity. That means
414  // that when starting with an indirectly allocated prevector with
415  // size and capacity > N, the result may be a still indirectly allocated
416  // prevector with size <= N and capacity > N. A shrink_to_fit() call is
417  // necessary to switch to the (more efficient) directly allocated
418  // representation (with capacity N and size <= N).
419  iterator p = first;
420  char* endp = (char*)&(*end());
421  if (!std::is_trivially_destructible<T>::value) {
422  while (p != last) {
423  (*p).~T();
424  _size--;
425  ++p;
426  }
427  } else {
428  _size -= last - p;
429  }
430  memmove(&(*first), &(*last), endp - ((char*)(&(*last))));
431  return first;
432  }
