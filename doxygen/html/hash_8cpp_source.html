20 {
21  // The following is MurmurHash3 (x86_32), see http://code.google.com/p/smhasher/source/browse/trunk/MurmurHash3.cpp
22  uint32_t h1 = nHashSeed;
23  if (vDataToHash.size() > 0)
24  {
25  const uint32_t c1 = 0xcc9e2d51;
26  const uint32_t c2 = 0x1b873593;
27 
28  const int nblocks = vDataToHash.size() / 4;
29 
30  //----------
31  // body
32  const uint8_t* blocks = &vDataToHash[0] + nblocks * 4;
33 
34  for (int i = -nblocks; i; i++) {
35  uint32_t k1 = ReadLE32(blocks + i * 4);
36 
37  k1 *= c1;
38  k1 = ROTL32(k1, 15);
39  k1 *= c2;
40 
41  h1 ^= k1;
42  h1 = ROTL32(h1, 13);
43  h1 = h1 * 5 + 0xe6546b64;
44  }
45 
46  //----------
47  // tail
48  const uint8_t* tail = (const uint8_t*)(&vDataToHash[0] + nblocks * 4);
49 
50  uint32_t k1 = 0;
51 
52  switch (vDataToHash.size() & 3) {
53  case 3:
54  k1 ^= tail[2] << 16;
55  case 2:
56  k1 ^= tail[1] << 8;
57  case 1:
58  k1 ^= tail[0];

60  k1 = ROTL32(k1, 15);

62  h1 ^= k1;
63  }
64  }
65

70  h1 *= 0x85ebca6b;
