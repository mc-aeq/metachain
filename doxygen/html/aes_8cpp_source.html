73 template <typename T>
74 static int CBCEncrypt(const T& enc, const unsigned char iv[AES_BLOCKSIZE], const unsigned char* data, int size, bool pad, unsigned char* out)
75 {
76  int written = 0;
77  int padsize = size % AES_BLOCKSIZE;
78  unsigned char mixed[AES_BLOCKSIZE];
79 
80  if (!data || !size || !out)
81  return 0;
82 
83  if (!pad && padsize != 0)
84  return 0;
85 
86  memcpy(mixed, iv, AES_BLOCKSIZE);
87 
88  // Write all but the last block
89  while (written + AES_BLOCKSIZE <= size) {
90  for (int i = 0; i != AES_BLOCKSIZE; i++)
91  mixed[i] ^= *data++;
92  enc.Encrypt(out + written, mixed);
93  memcpy(mixed, out + written, AES_BLOCKSIZE);
94  written += AES_BLOCKSIZE;
95  }
96  if (pad) {
97  // For all that remains, pad each byte with the value of the remaining
98  // space. If there is none, pad by a full block.
99  for (int i = 0; i != padsize; i++)
100  mixed[i] ^= *data++;
101  for (int i = padsize; i != AES_BLOCKSIZE; i++)
102  mixed[i] ^= AES_BLOCKSIZE - padsize;
103  enc.Encrypt(out + written, mixed);
104  written += AES_BLOCKSIZE;
105  }
106  return written;
107 }

109 template <typename T>
110 static int CBCDecrypt(const T& dec, const unsigned char iv[AES_BLOCKSIZE], const unsigned char* data, int size, bool pad, unsigned char* out)
111 {
112  int written = 0;
113  bool fail = false;
114  const unsigned char* prev = iv;
115 
116  if (!data || !size || !out)
117  return 0;
118 
119  if (size % AES_BLOCKSIZE != 0)
120  return 0;
121 
122  // Decrypt all data. Padding will be checked in the output.
123  while (written != size) {
124  dec.Decrypt(out, data + written);
125  for (int i = 0; i != AES_BLOCKSIZE; i++)
126  *out++ ^= prev[i];
127  prev = data + written;
128  written += AES_BLOCKSIZE;
129  }
130 
131  // When decrypting padding, attempt to run in constant-time
132  if (pad) {
133  // If used, padding size is the value of the last decrypted byte. For
134  // it to be valid, It must be between 1 and AES_BLOCKSIZE.
135  unsigned char padsize = *--out;
136  fail = !padsize | (padsize > AES_BLOCKSIZE);
137 
138  // If not well-formed, treat it as though there's no padding.
139  padsize *= !fail;
140 
141  // All padding must equal the last byte otherwise it's not well-formed
142  for (int i = AES_BLOCKSIZE; i != 0; i--)
143  fail |= ((i > AES_BLOCKSIZE - padsize) & (*out-- != padsize));
144 
145  written -= padsize;
146  }
147  return written * !fail;
148 }
