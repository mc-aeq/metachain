141 CSHA256& CSHA256::Write(const unsigned char* data, size_t len)
142 {
143  const unsigned char* end = data + len;
144  size_t bufsize = bytes % 64;
145  if (bufsize && bufsize + len >= 64) {
146  // Fill the buffer, and process it.
147  memcpy(buf + bufsize, data, 64 - bufsize);
148  bytes += 64 - bufsize;
149  data += 64 - bufsize;
150  sha256::Transform(s, buf);
151  bufsize = 0;
152  }
153  while (end >= data + 64) {
154  // Process full chunks directly from the source.
155  sha256::Transform(s, data);
156  bytes += 64;
157  data += 64;
158  }
159  if (end > data) {
160  // Fill the buffer with what remains.
161  memcpy(buf + bufsize, data, end - data);
162  bytes += end - data;
163  }
164  return *this;
165 }
