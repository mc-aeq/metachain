29 void CScheduler::serviceQueue()
30 {
31  boost::unique_lock<boost::mutex> lock(newTaskMutex);
32  ++nThreadsServicingQueue;
33 
34  // newTaskMutex is locked throughout this loop EXCEPT
35  // when the thread is waiting or when the user's function
36  // is called.
37  while (!shouldStop()) {
38  try {
39  if (!shouldStop() && taskQueue.empty()) {
40  reverse_lock<boost::unique_lock<boost::mutex> > rlock(lock);
41  // Use this chance to get a tiny bit more entropy
42  RandAddSeedSleep();
43  }
44  while (!shouldStop() && taskQueue.empty()) {
45  // Wait until there is something to do.
46  newTaskScheduled.wait(lock);
47  }
48 
49  // Wait until either there is a new task, or until
50  // the time of the first item on the queue:
51 
52  // wait_until needs boost 1.50 or later; older versions have timed_wait:
53 #if BOOST_VERSION < 105000
54  while (!shouldStop() && !taskQueue.empty() &&
55  newTaskScheduled.timed_wait(lock, toPosixTime(taskQueue.begin()->first))) {
56  // Keep waiting until timeout
57  }
58 #else
59  // Some boost versions have a conflicting overload of wait_until that returns void.
60  // Explicitly use a template here to avoid hitting that overload.
61  while (!shouldStop() && !taskQueue.empty()) {
62  boost::chrono::system_clock::time_point timeToWaitFor = taskQueue.begin()->first;
63  if (newTaskScheduled.wait_until<>(lock, timeToWaitFor) == boost::cv_status::timeout)
64  break; // Exit loop after timeout, it means we reached the time of the event
65  }
66 #endif
67  // If there are multiple threads, the queue can empty while we're waiting (another
68  // thread may service the task we were waiting on).
69  if (shouldStop() || taskQueue.empty())
70  continue;
71 
72  Function f = taskQueue.begin()->second;
73  taskQueue.erase(taskQueue.begin());
74 
75  {
76  // Unlock before calling f, so it can reschedule itself or another task
77  // without deadlocking:
78  reverse_lock<boost::unique_lock<boost::mutex> > rlock(lock);
79  f();
80  }
81  }
82  catch (...) {
83  --nThreadsServicingQueue;
84  throw;
85  }
86  }
87  --nThreadsServicingQueue;
88  newTaskScheduled.notify_one();
89 }
